function aiEnhance() { const bg = canvas.backgroundImage; if (!bg) { alert("Upload an image first"); return; } bg.filters = [ new fabric.Image.filters.Brightness({ brightness: 0.1 }), new fabric.Image.filters.Contrast({ contrast: 0.25 }), new fabric.Image.filters.Saturation({ saturation: 0.3 }), new fabric.Image.filters.Sharpen() ]; bg.applyFilters(); canvas.renderAll(); }
